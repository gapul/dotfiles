│   │   ├── platform-detection.nix
│   │   ├── themes/colors.nix  
│   │   └── security/sops/
│   ├── darwin/             # macOS特化
│   ├── linux/              # Linux特化  
│   ├── wsl/                # WSL特化
│   └── android/            # Android特化
